изменение массива - dinosaurs[0] = "Тираннозавр рекс";

mas.length; - Возвращает длину массива mas
mas[mas.length -1]; - возвращает последний элемент массива

mas.push("lama"); - добавляет эл lama в конец массива
mas.unshift("hello") - добавляет эл hello в начало массива

mas.pop() - удаляет последний элемент и выводит
mas.shift() - удаляет первый элемент и выводит

mas.concat(mas2, mas3) - соединяет элементы массивов mas+mas2+mas3

mas.indexOf("элемент") - воз-ет индекс элемента, if элемент не найден воз-ет (-1)

mas.join(); - соединяет все элементы в одну строку
mas.join(" "); - |-|-| и разделяет элементы пробелом

str.split() - строку str превращает в массив
